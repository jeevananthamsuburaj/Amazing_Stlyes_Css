<script>
const navbar = document.querySelector(".navbar");
const items = document.querySelectorAll(".navbar li");
const mountain = document.querySelector(".mountain");
const stone = document.querySelector(".stone");

/* find mountain X position under li */
function getMountainX(li) {
  const rect = li.getBoundingClientRect();
  const nav = navbar.getBoundingClientRect();
  return rect.left - nav.left + rect.width/2 - mountain.offsetWidth/2;
}

/* place mountain initially */
mountain.style.transform = `translateX(${getMountainX(items[0])}px)`;

/* click event */
items.forEach(li => {
  li.addEventListener("click", () => {

    /* remove active */
    items.forEach(x => x.classList.remove("active"));

    const nav = navbar.getBoundingClientRect();
    const icon = li.querySelector("img").getBoundingClientRect();

    /* ---- 1. slide mountain under li ---- */
    let mx = getMountainX(li);
    mountain.style.transform = `translateX(${mx}px)`;

    /* ---- 2. mountain shoot-up small effect ---- */
    mountain.animate([
      { transform:`translateX(${mx}px) translateY(0)` },
      { transform:`translateX(${mx}px) translateY(-25px)` },
      { transform:`translateX(${mx}px) translateY(0)` }
    ],{ duration:350, easing:"ease-out" });

    /* ---- 3. STONE START POSITION (mountain top center) ---- */
    const mRect = mountain.getBoundingClientRect();
    const startX = mRect.left - nav.left + mRect.width/2 - 7;
    const startY = mRect.top - nav.top - 5;   // small GAP above mountain

    /* ---- 4. STONE END POSITION (icon center) ---- */
    const iconX = icon.left - nav.left + icon.width/2 - 7;
    const iconY = icon.top - nav.top + icon.height/2 - 7;

    /* set stone start position */
    stone.style.left = startX + "px";
    stone.style.top = startY + "px";
    stone.style.display = "block";

    /* stone travel animation */
    const anim = stone.animate([
      { transform:"translate(0,0) scale(1)" },
      { transform:`translate(${iconX-startX}px, ${iconY-startY}px) scale(0.9)` }
    ],{
      duration:400,
      easing:"ease-out",
      fill:"forwards"
    });

    /* ---- 5. when stone reaches icon â†’ hide + color change ---- */
    anim.onfinish = () => {
      stone.style.display="none";
      li.classList.add("active");
    };

  });
});
</script>

<script>
const navbar = document.querySelector(".navbar");
const items = document.querySelectorAll(".navbar li");
const mountain = document.querySelector(".mountain");
const stone = document.querySelector(".stone");

// get mountain X under li
function getMountainX(li) {
  const rect = li.getBoundingClientRect();
  const nav = navbar.getBoundingClientRect();
  return rect.left - nav.left + rect.width/2 - mountain.offsetWidth/2;
}

// place mountain initially
mountain.style.transform = `translateX(${getMountainX(items[0])}px)`;

items.forEach(li => {
  li.addEventListener("click", () => {

    // remove active from all
    items.forEach(x => x.classList.remove("active"));

    const nav = navbar.getBoundingClientRect();
    const icon = li.querySelector("img").getBoundingClientRect();

    // move mountain horizontally
    const mx = getMountainX(li);

    // mountain jump animation
    const jumpHeight = -25; // negative = up
    const mountainAnim = mountain.animate([
      { transform:`translateX(${mx}px) translateY(0)` },
      { transform:`translateX(${mx}px) translateY(${jumpHeight}px)` },
      { transform:`translateX(${mx}px) translateY(0)` }
    ], {
      duration: 400,
      easing: "ease-out",
      fill: "forwards"
    });

    // after jump, set final transform
    mountainAnim.onfinish = () => {
      mountain.style.transform = `translateX(${mx}px) translateY(0)`;
    };

    // stone start (mountain top center)
    const mRect = mountain.getBoundingClientRect();
    const startX = mRect.left - nav.left + mRect.width/2 - 7;
    const startY = mRect.top - nav.top - 5; // small gap above mountain

    // stone end (icon center)
    const endX = icon.left - nav.left + icon.width/2 - 7;
    const endY = icon.top - nav.top + icon.height/2 - 7;

    stone.style.left = startX + "px";
    stone.style.top = startY + "px";
    stone.style.display = "block";

    stone.animate([
      { transform:"translate(0,0) scale(1)", opacity:1 },
      { transform:`translate(${endX-startX}px, ${endY-startY}px) scale(0.9)`, opacity:0 }
    ], { duration:400, easing:"ease-out", fill:"forwards" })
    .onfinish = () => {
      stone.style.display="none";
      li.classList.add("active");
    };
  });
});
</script>
